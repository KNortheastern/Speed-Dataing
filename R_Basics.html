# C = (F - 32) / 1.8
tempF <- 57
(tempF - 32) / 1.8


# function to convert F to C
fahr_to_celsius <- function(randomVariableNAme) {
  tempC <- (randomVariableNAme - 32) / 1.8
  return(tempC)
}
fahr_to_celsius(57)
fahr_to_celsius(tempF)
